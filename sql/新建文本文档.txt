CREATE TABLE Adminitor(
  admin_id  INT AUTO_INCREMENT,
  admin_name VARCHAR(20),
  user_id INT,
  admin_phone INT,
  PRIMARY KEY(admin_id)
  )CHARSET=utf8;

CREATE TABLE Groups(
  groups_id INT AUTO_INCREMENT,
  groups_name VARCHAR(20),
  groups_direct VARCHAR(50),
  gl_id INT,
  PRIMARY KEY(groups_id )
)CHARSET=utf8;
