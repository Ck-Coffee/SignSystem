CREATE  TABLE  Users(
    user_id  INT  AUTO_INCREMENT,
    user_name  VARCHAR(20),
    user_phone  VARCHAR(20),
    user_qq  VARCHAR(20),
    user_email  VARCHAR(20),
    user_sex  ENUM('男','女'),
    user_birthday  DATE,
    user_image  TINYTEXT,
    group_id  INT  DEFAULT  0,
    PRIMARY  KEY(user_id)
)CHARSET=utf8;

CREATE  TABLE  User_Password(
    pwd_id  INT  AUTO_INCREMENT,
    user_pwd  VARCHAR(200),
    user_id  INT,
    PRIMARY  KEY(pwd_id)
)CHARSET=utf8;

CREATE  TABLE  Salts(
    salt_id  INT  AUTO_INCREMENT,
    salt  VARCHAR(10),
    pwd_id  INT,
    PRIMARY  KEY(salt_id)
)CHARSET=utf8;
