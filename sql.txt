CREATE TABLE users(
	user_id INT AUTO_INCREMENT,
	user_name VARCHAR(20),
	user_phone VARCHAR(12),
	user_qq INT,
	user_email VARCHAR(20),
	user_sex ENUM('男','女'),
	user_birthday DATE,
	user_image TINYTEXT,
	group_id INT UNIQUE,
	PRIMARY KEY(user_id)
) ENGINE=INNODB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8;


CREATE TABLE Pwd(
	pwd_id INT AUTO_INCREMENT,
	passwords VARCHAR(200),
	user_id INT UNIQUE,
	PRIMARY KEY(pwd_id)
)ENGINE=INNODB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8;



CREATE TABLE salt(

	salt_id INT AUTO_INCREMENT,
	salt VARCHAR(10),
	pwd_id INT,
	PRIMARY KEY(salt_id)
)ENGINE=INNODB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8;
