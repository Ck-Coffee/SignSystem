CREATE TABLE Sign_in(
  user_id INT ,
  sign_date DATE,
  user_name VARCHAR(10) ,
  sign_status ENUM('已签到','迟签到','未签到'),
  sign_time TIME,
  PRIMARY KEY(user_id , sign_date)
)CHARSET=utf8;

CREATE TABLE Questions(
  questions_id INT AUTO_INCREMENT,
  questions_content VARCHAR(50),
  questions_answer VARCHAR(20),
  PRIMARY KEY(questions_id)
)CHARSET=utf8;

CREATE TABLE News(
	news_id INT AUTO_INCREMENT PRIMARY KEY,
	news_datetime TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
	news_watchnums INT DEFAULT  0,
	news_filename TEXT,
	news_supports INT DEFAULT  0,
	news_dissuport INT DEFAULT  0
)CHARSET=utf8;

CREATE TABLE Messages(
  message_id INT AUTO_INCREMENT,
  message_datetime TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  message_mark ENUM('1','2','3'),
  group_id  INT,
  message_content VARCHAR(50),
  PRIMARY KEY(message_id)
)CHARSET=utf8;

CREATE  TABLE  Users(
    user_id  INT  AUTO_INCREMENT,
    user_name  VARCHAR(20),
    user_phone  VARCHAR(20),
    user_qq  VARCHAR(20),
    user_email  VARCHAR(20),
    user_sex  ENUM('男','女'),
    user_birthday  DATE,
    user_image  TINYTEXT,
    group_id  INT  DEFAULT  0 ,
    PRIMARY  KEY(user_id)
)CHARSET=utf8;

CREATE TABLE User_Password(
  pwd_id INT AUTO_INCREMENT,
  user_pwd VARCHAR(200),
  user_id INT,
  PRIMARY KEY(pwd_id)
)CHARSET=utf8;

CREATE TABLE Salts(
  salt_id INT AUTO_INCREMENT,
  salt VARCHAR(10),
  pwd_id INT,
  PRIMARY KEY(salt_id)
)CHARSET=utf8;

CREATE TABLE Adminitors(
  admin_id  INT AUTO_INCREMENT,
  admin_name VARCHAR(20),
  user_id INT,
  admin_phone INT,
  PRIMARY KEY(admin_id)
  )CHARSET=utf8;
  
CREATE TABLE Groups(
  group_id INT AUTO_INCREMENT,
  group_name VARCHAR(20),
  group_direct VARCHAR(50),
  groupleader_id INT,
  PRIMARY KEY(groups_id )
)CHARSET=utf8;


CREATE TABLE Posts (
  post_id INT AUTO_INCREMENT,
  post_datetime TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  user_id INT,
  user_name VARCHAR(20),
  post_supports INT DEFAULT 0,
  post_comments INT DEFAULT 0,
  post_content_file VARCHAR(30),
  CONSTRAINT PK_Post_id PRIMARY KEY(post_id)
)CHARSET=utf8;

CREATE TABLE Comments(
  comment_id INT AUTO_INCREMENT,
  post_id INT,
  user_id INT,
  parent_comment_id INT,
  comment_datetime TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  comment_content_file VARCHAR(30),
  CONSTRAINT PK_Comment_id PRIMARY KEY(comment_id)
)CHARSET=utf8;
